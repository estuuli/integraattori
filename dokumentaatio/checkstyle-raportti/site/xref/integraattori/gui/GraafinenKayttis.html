19  /**
20   * Graafinen käyttöliittymä.

24  public class GraafinenKayttis implements Runnable {
25  
26      private JFrame frame;
27  
28      /**
29       * Luokan konstruktori.
30       *
31       */
32      public GraafinenKayttis() {
33      }
34  
35      @Override
36      public void run() {
37  
38          frame = new JFrame("Integraattori");
39          frame.setPreferredSize(new Dimension(500, 700));
40  
41          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
42  
43          luoKomponentit(frame.getContentPane());
44  
45          frame.pack();
46          frame.setVisible(true);
47          //throw new UnsupportedOperationException("Not supported yet."); //To change body of generated methods, choose Tools | Templates.
48      }
49  
50      /**
51       * Luo graafisen käyttöliittymän komponenetit.
52       *
53       */
54      private void luoKomponentit(Container container) {
55  
56          BoxLayout layout = new BoxLayout(container, BoxLayout.Y_AXIS);
57          container.setLayout(layout);
58  
59          container.add(new JLabel("Valitse käytettävä integrointimenetelmä:"));
60          
61          MetodinValinnanKuuntelija metodiKuuntelija = new MetodinValinnanKuuntelija();
62          
63  //        JRadioButton kulmio = new JRadioButton("Suorakulmiometodi");
64  //        kulmio.setActionCommand("suorakulmio");
65  //        kulmio.addActionListener(metodiKuuntelija);
66          JRadioButton suunnikas = new JRadioButton("Puolisuunnikasmetodi");
67          suunnikas.setActionCommand("puolisuunnikas");
68          suunnikas.addActionListener(metodiKuuntelija);
69          JRadioButton simpson = new JRadioButton("Simpsonin metodi");
70          simpson.setActionCommand("simpson");
71          simpson.addActionListener(metodiKuuntelija);
72          //simpson.setSelected(true);
73  
74          ButtonGroup buttonGroup = new ButtonGroup();
75  //        buttonGroup.add(kulmio);
76          buttonGroup.add(suunnikas);
77          buttonGroup.add(simpson);
78  
79          JLabel funktioTeksti = new JLabel("Funktio: ");
80          JTextField funktioKentta = new JTextField();
81  //        JLabel muuttujaTeksti = new JLabel("Muuttuja: ");
82  //        JTextField muuttujaKentta = new JTextField();
83          JLabel alarajaTeksti = new JLabel("Alaraja: ");
84          JTextField alarajaKentta = new JTextField();
85          JLabel ylarajaTeksti = new JLabel("Yläraja: ");
86          JTextField ylarajaKentta = new JTextField();
87          JLabel tarkkuusTeksti = new JLabel("Haluttu tarkkuus: ");
88          JTextField tarkkuusKentta = new JTextField();
89          JLabel virheilmoitusTeksti = new JLabel("Virheilmoitus: ");
90          JTextField virheilmoitusKentta = new JTextField();
91  
92          
93          
94          JButton integroijaNappi = new JButton("Integroi!");       
95  
96          JLabel tulosTeksti = new JLabel("Tulos: ");
97          JTextField tulosKentta = new JTextField();
98          JLabel virheTeksti = new JLabel("Virhe: ");
99          JTextField virheKentta = new JTextField();
100 
101         ParametrienLisaysKuuntelija parametriKuuntelija = new ParametrienLisaysKuuntelija(funktioKentta, alarajaKentta, ylarajaKentta, tarkkuusKentta, virheilmoitusKentta, tulosKentta, virheKentta, metodiKuuntelija);
102         integroijaNappi.addActionListener(parametriKuuntelija);
103         
104 //        container.add(kulmio);
105         container.add(suunnikas);
106         container.add(simpson);
107         container.add(funktioTeksti);
108         container.add(funktioKentta);
109 //        container.add(muuttujaTeksti);
110 //        container.add(muuttujaKentta);
111         container.add(alarajaTeksti);
112         container.add(alarajaKentta);
113         container.add(ylarajaTeksti);
114         container.add(ylarajaKentta);
115         container.add(tarkkuusTeksti);
116         container.add(tarkkuusKentta);
117         container.add(integroijaNappi);
118         container.add(virheilmoitusTeksti);
119         container.add(virheilmoitusKentta);
120 
121         container.add(tulosTeksti);
122         container.add(tulosKentta);
123         container.add(virheTeksti);
124         container.add(virheKentta);
125     }
126 
127     public JFrame getFrame() {
128         return frame;
129     }
